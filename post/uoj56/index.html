总览

目前 $90$ 分，容易做到 $91$ 但是目前没必要。

基本要求：$0\le n\le10000$，$0\le m\le20000$，$0\le w\le20000$。无重边，可以有自环。

$10$ 个测试点的任务和更多限制：

  • test 1：邻接矩阵。$k=0$，$n\le500$。
  • test 2：单源最短路。$k=10$。
  • test 3：全源最短路（似乎有锅）。$k=11$，$n\le100$。
  • test 4：单源次短路。$k=12$。
  • test 5：强连通分量。$k=20$。
  • test 6：集合划分哈希。$k=21$，$n\le10$
  • test 7：集合划分单调性哈希。$k=22$，$n\le20$。
  • test 8：二分图最大匹配。$k=30$，$n\le100$。
  • test 9：二分图最大匹配，某种实现的匈牙利算法构造。$k=31$，$n\le20$。
  • test 10：SPFA 过程中的队列。$k=13$，$n\le500$，$m\le3000$。

要对这些东西反推。$p$ 在大多数测试点表示边的数目，test 1,5,10 除外：test 1 中总有 $p=0$，test 5 中 $p=\text{强连通分量数目}$，test 10 中 $p=\text{队列长度}$。

 1uint Dist[20005],P[20005];
 2int main()

 4    freopen("nm2.in","r",stdin);

 6    uint n,k,p;scanf("%u%u%u",&n,&k,&p);

 8    std::sort(P,P+n,[&](uint i,uint j){return Dist[i]<Dist[j];});

10    for(uint i=1;i<n;i++)
11        printf("%u %u %u\n",P[i-1]+1,P[i]+1,Dist[P[i]]-Dist[P[i-1]]);
12    p-=n-1;
13    for(uint i=1;i<n;i++)for(uint j=0;p&&j<i;j++)
14        printf("%u %u 20000\n",P[i]+1,P[j]+1),p--;

test 3

把所有不能松弛的边找出来即可。

注意这样恰好 $398$ 条边，比较一下会发现少两个特殊的自环，补上即可。

 1uint Dist[405][405];
 2bol B[405][405];
 3int main()
 4{
 5    freopen("nm3.in","r",stdin);
 6    freopen("nm3.out","w",stdout);
 7    uint n,k,p;scanf("%u%u%u",&n,&k,&p);
 8    printf("%u %u %u\n",n,p,k);
 9    for(uint i=0;i<n;i++)for(uint j=0;j<n;j++)scanf("%u",Dist[i]+j);
10    for(uint i=0;i<n;i++)for(uint j=0;j<n;j++)if(Dist[i][j])
11    {
12        B[i][j]=true;
13        for(uint k=0;k<n;k++)if(i!=k&&j!=k&&Dist[i][k]+Dist[k][j]==Dist[i][j])
14        {
15            B[i][j]=false;break;
16        }
17        if(B[i][j])
18            printf("%u %u %u\n",i+1,j+1,Dist[i][j]);
19    }
20    return 0;
21}

test 6

注意到点数边数均为 $8$，直接对拍搞出所有集合划分即可。

 1FILE*fin,*fout;
 2uint P[15],n,p;
 3std::vector<std::vector<uint> >V;
 4voi dfs(uint p)
 5{
 6    if(p==n)
 7    {
 8        V.push_back(std::vector<uint>(P,P+n));
 9        return;
10    }
11    for(uint&e=P[p]=0;e<=p;e++)if(P[e]==e)
12        dfs(p+1);
13}
14int main()
15{
16    fin=fopen("nm6.in","r");
17    uint k;fscanf(fin,"%u%u%u",&n,&k,&p);
18    fclose(fin);
19    dfs(0);
20    uint j=0;
21    while(j<V.size())
22    {
23        fout=fopen("nm6.out","w");
24        fprintf(fout,"%u %u %u\n",n,p,k);
25        std::vector<uint>T[15];
26        for(uint i=0;i<n;i++)
27            T[V[j][i]].push_back(i);
28        uint p0=p;
29        for(uint i=0;i<n;i++)if(T[i].size()>1)
30        {
31            for(uint j=1;j<T[i].size();j++)
32                fprintf(fout,"%u %u 0\n",T[i][j-1]+1,T[i][j]+1),p0--;
33            fprintf(fout,"%u %u 0\n",T[i].back()+1,i+1),p0--;
34        }
35        for(uint i=0;i<p0;i++)
36            fprintf(fout,"%u %u 0\n",i+1,i+1);
37        fclose(fout);
38        system("./code nm6.out > QAQ.err");
39        if(!system("diff QAQ.err nm6.in"))
40            break;
41        j++;
42    }
43    return 0;
44}

test 9

测试一下容易发现这个匈牙利是依次选还没匹配的，再往前增广。

那选当前对应的，连前面对应的，连更大还没连过的即可。

测了一下刚刚好。

 1uint P[105];
 2int main()
 3{
 4    freopen("nm9.in","r",stdin);
 5    freopen("nm9.out","w",stdout);
 6    uint n,k,p;scanf("%u%u%u",&n,&k,&p);
 7    for(uint i=0;i<n;i++)scanf("%u",P+i),P[i]--;
 8    std::vector<std::pair<uint,uint> >V;
 9    for(uint i=0;i<n;i++)V.push_back({i,P[i]}),p--;
10    for(uint i=0;i<n;i++)
11    {
12        static bol B[105];
13        for(uint j=0;j<n;j++)B[j]=true;
14        for(uint j=0;p&&j<i;j++)
15            V.push_back({i,P[j]}),p--,B[P[j]]=false;
16        for(uint j=P[i]+1;p&&j<n;j++)if(B[j])
17            V.push_back({i,j}),p--;
18    }
19    printf("%u %u %u\n",n,(uint)V.size(),k);
20    for(auto s:V)
21        printf("%u %u 0\n",s.first+1,s.second+1);
