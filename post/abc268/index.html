 1typedef std::vector<chr>str;
 2std::set<str>S;str C[32],now;chr User[32];uint n;bol Used[32];
 3voi dfs(uint p){
 4    if(now.size()>16)return;
 5    if(p==n){
 6        for(uint i=0;i<n;i++)if(!Used[i]){
 7            now.insert(now.end(),C[i].begin(),C[i].end());
 8            if(now.size()>=3&&now.size()<=16&&!S.count(now)){
 9                for(auto s:now)putchar(s);
10                exit(0);
11            }
12            now.erase(now.end()-C[i].size(),now.end());
13        }
14        return;
15    }
16    uint lim=18-now.size();
17    for(uint i=0;i<n;i++)if(!Used[i])lim-=C[i].size()+1;
18    for(uint i=0;i<n;i++)if(!Used[i]){
19        Used[i]=1,now.insert(now.end(),C[i].begin(),C[i].end());
20        for(uint k=1;k<=lim;k++)
21        {
22            for(uint t=1;t<=k;t++)now.push_back('_');
23            dfs(p+1);
24            now.erase(now.end()-k,now.end());
25        }
26        now.erase(now.end()-C[i].size(),now.end()),Used[i]=0;
27    }
28}

E

还是考虑计算差量。

反向考虑对每种旋转方案的贡献，然后就是区间加一次函数。

直接用差分维护一下即可。

细节有亿点多。

 1ullt A[200005],B[200005];uint n;
 2voi add_base(ullt*A,uint l,uint r,ullt v){A[r]-=v,A[l]+=v;}
 3voi add(uint l,uint r,ullt k,ullt b){add_base(A,l,r,k),add_base(B,l,r,b);}
 4voi add_all(int l,int r,ullt k,ullt b){
 5    if(l<0)
 6    {
 7        add(l+n,n,k,b-n*k),add(0,r,k,b);
 8        return;
 9    }
10    if(r>(int)n){
11        add(l,n,k,b),add(0,r-n,k,b+k*n);
12        return;
13    }
14    add(l,r,k,b);
15}
16int main()
17{
18#ifdef MYEE
19    freopen("QAQ.in","r",stdin);

24        scanf("%u",&v),v=(v+n-i)%n;
25        add_all(v-n/2,v,-1llu,v);
26        add_all(v,v+n-n/2,1llu,-(ullt)v);
27    }
28    for(uint i=0;i<n;i++)A[i+1]+=A[i],B[i+1]+=B[i];

G

考虑对排名反向计算贡献。

对 $i$ 在 $j$ 前的概率，有以下几种状态：

  • $s_i$ 为 $s_j$ 前缀，则 $i$ 必然在 $j$ 前。
  • $s_j$ 为 $s_i$ 前缀，则 $i$ 必然在 $j$ 后。
  • 其余情况，前后顺序等可能出现。

直接拿棵 Trie 计算一下每个串有几个前缀出现在全集中，及作为全集中多少串的前缀。

然后随便统计一下信息就完了。

 1const ullt Mod=998244353;
 2typedef ConstMod::mod_ullt<Mod>modint;
 3struct node{node*son[26];uint v1,v2;}N[2000005],*rot=N,*cnt=N+1;
 4node*NewNode(){return cnt++;}
 5#define Id(p) ((uint)((p)-N))
 6#define Turn(c) ((c)-'a')
 7node*insert(chr*C){
 8    node*p=N;
 9    while(*C){
10        p->v1++;auto&s=p->son[Turn(*(C++))];
11        if(!s)s=NewNode();
12        p=s;
13    }
14    p->v2++;
15    return p;
16}
17chr C[500005];
18node*At[500005];
19int main()
20{
21#ifdef MYEE
22    freopen("QAQ.in","r",stdin);
23    // freopen("QAQ.out","w",stdout);
24#endif
25    uint n;scanf("%u",&n);
26    for(uint i=0;i<n;i++)scanf("%s",C),At[i]=insert(C);
27    for(node*p=rot;p!=cnt;p++)for(auto s:p->son)if(s)s->v2+=p->v2;
28    for(uint i=0;i<n;i++){
29        (modint(n-At[i]->v1+At[i]->v2)/2).println();
30    }

32}

H

考虑对文本串每个前缀，计算出其最短的存在于模板串中的后缀。

这个拿个 ACAM 即可搞定。

然后就是每个这样的区间都得删掉一个点，直接贪心 / 差分约束即可。

 1const uint sigma=26;
 2struct node{node*son[sigma],*fail;uint kill;node():kill(-1u){}}N[2000005],*rot=N,*cnt=N+1;
 3node*NewNode(){return cnt++;}
 4#define Id(p) ((uint)((p)-N))
 5#define Turn(c) ((c)-'a')
 6voi insert(chr*C){
 7    node*p=N;uint len=0;
 8    while(*C){
 9        auto&s=p->son[Turn(*(C++))];
10        if(!s)s=NewNode();
11        p=s,len++;
12    }
13    _min(p->kill,len);
14}
15voi build(){
16    std::queue<node*>Q;rot->fail=rot;
17    for(uint i=0;i<sigma;i++)(rot->son[i]?(Q.push(rot->son[i]),rot->son[i]->fail):rot->son[i])=rot;
18    while(!Q.empty()){
19        node*p=Q.front();Q.pop(),_min(p->kill,p->fail->kill);
20        for(uint i=0;i<sigma;i++)
21            (p->son[i]?(Q.push(p->son[i]),p->son[i]->fail):p->son[i])=p->fail->son[i];
22    }
23}
24chr S[500005],C[500005];
25int main()
26{
27#ifdef MYEE
28    freopen("QAQ.in","r",stdin);
29    // freopen("QAQ.out","w",stdout);
30#endif
31    scanf("%s",S);
32    uint n;scanf("%u",&n);
33    while(n--){
34        scanf("%s",C);
35        insert(C);
36    }
37    build();
38    uint ans=0,last=1e9;
39    node*p=rot;
40    for(auto c:S)if(c){
41        last++,p=p->son[Turn(c)];
42        if(last>=p->kill)
43            last=0,ans++;
44    }else break;
45    printf("%u\n",ans);
46    return 0;
47}
